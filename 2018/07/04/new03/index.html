Mysql 基本操作

发表于 2018-07-04 |
字数统计: 288 | 阅读时长 ≈ 1

分组查询某个字段的所有值

select gradeName,group_concat(stuName) from student group by gradeName;

分组查询各组个数

select gradeName,count(stuName) from student group by gradeName;

having查询个数大于3

select gradeName,count(stuName) from student group by gradeName having count(stuName)>3;

分页查询limit

select * from student limit 0,5;#从0开始取5个

聚合函数(默认和group by一起使用)

count()

select count(*) as total from student;#as total取别名

sum()

select sum(grade) from student group by stuName;

或者执行如下语句：

select sum(grage) from student where stuName="张三";

其他聚合函数：avg(),max(),min()

加法中存在null的处理方法

select salary*12+ifnull(bouns,0) from emp;#bouns奖金,如果为null则用0替换

笛卡尔乘积(如：A表3条数据,B表4条数据,合并为12条数据<所有情况>)

select * from student,score;

内连接

select * from student,score where student.id=score.id;

外连接

左连接(查询左表的所有数据)

select * from student left join score on student.id=score.id;

右连接(查询右表的所有数据)

select * from student right join score on student.id=score.id;

多条件查询(where 后的多个条件用 and 连接)

select * from student right join score on student.id=score.id and student.id>3;
