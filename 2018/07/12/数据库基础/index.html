数据库基础

发表于 2018-07-12 |
字数统计: 631 | 阅读时长 ≈ 3

子查询

带In关键字(IN,NOT IN)

select * from t_book where booktypeId in (select id from t_booktype);

带比较运算符(括号中最好为单个值)

select * from t_book where price>=(select price from t_priceLevel where priceLevel=1);

带Exists关键字(EXISTS,NOT EXISTS)

select from t_book where exists (select from t_booktype);

带Any关键字

select * from t_book where price>=ANY (select price from t_priceLevel);#any后为集合,选择大于其中任何一个

带All关键字

select * from t_book where price>=ALL (select price from t_priceLevel);#选取满足所有条件的

合并查询结果

UNION(去除相同的结果)

select id from t_book UNION select id from t_booktype;

UNION ALL(不去除相同的结果)

select id from t_book UNION ALL select id from t_booktype;

为表和字段取别名

为表取别名

select * from t_book t where t.id=1;#别名t

为字段取别名(as关键字)

select t.bookName AS bname from t_book where t.id=1;

插入数据

insert into t_book(bookName,author) VALUES (“hello”,”lee”),(“hello1”,”lee1”),(“hello2”,”lee2”);#多条数据逗号隔开

更新数据

update t_book SET bookName=”world” where bookName LIKE “%hello%”;#set后为更改后的信息

删除数据DELETE

delete from t_book where id=5;

索引

1.提高查询速度
2.创建和维护索引的时间增加了(内容改变索引也要改)

索引分类

普通索引，唯一性索引(UNIQUE)，全文索引(FULLTEXT,MYSQL默认引擎不支持)，
单列索引，多列索引，空间索引(SPATIAL)

创建索引

单列索引

普通索引(INDEX|KEY)

create table t_user1(id int,userName varchar(20),password varchar(20),INDEX (userName));

唯一性索引(主键默认)

create table t_user2(id int,userName varchar(20),password varchar(20),UNIQUE INDEX index_userName(userName));#index后为取的别名

多列索引

create table t_user3(id int,userName varchar(20),password varchar(20),INDEX index_userName_password(userName,password));

在已经存在的表中添加索引

create INDEX index_userName ON t_user4(userName);#单列
create UNIQUE INDEX index_userName ON t_user4(userName);
create INDEX index_userName_password ON t_user4(userName,password);#多列
ALTER table t_user5 ADD INDEX index_userName (userName);#alter创建单列索引
ALTER table t_user5 ADD INDEX index_userName_password (userName,password);#alter创建多列索引

删除索引

drop INDEX index_userName ON t_user5;

视图

虚拟的表，使操作简便化，增加数据的安全性，提高表的逻辑独立性

创建视图

CREATE [ALGORITHM={UNDEFIEND|MERGE|TEMPTABLE}] VIEW 视图名 [(属性清单)] AS SELECT 语句 [WITH [CASCADED|LOCAL] CHECK OPTION];

单表上创建视图

creat VIEW v1(b,p) AS select bookName,price from t_book;#b\p 给变量重命名
select * from v1;#查询v1中booName和price信息

多表上创建视图

create VIEW v2 AS select tb.bookName,tbu.bookTypeName from t_book tb,t_booktype tby where tb.bookTypeId=tby.id;

查看视图

DESC v2;
SHOW table status like “v2\t_book”;#状态,v2为续表
SHOW CREATE VIEW v2;#查看详细信息
